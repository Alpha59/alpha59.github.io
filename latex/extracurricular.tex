    {

            %<$ for highlight in org.highlights $>

                %<$ else $>
                    \item{<! highlight !>}
                %<$ endif $>
                %<$ if highlight.additionalDescriptions $>
                    \begin{cvitemssub}
                    %<$ for ad in highlight.additionalDescriptions $>
                        \item{<! ad !>}
                    %<$ endfor $>
                    \end{cvitemssub}
                %<$ endif $>
            %<$ endfor $>
          \end{cvitems}
        %<$ endif $>
    }
 <$ endfor $>
